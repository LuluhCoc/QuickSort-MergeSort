#define TAM 100

typedef struct {
	char nome[TAM], cargo[TAM], salario[TAM];
	int idade, cpf;
} cadastro;


cadastro dados[TAM], aux;
int i, j, quant;
void Quick(cadastro dados[TAM], int inicio, int fim);

void cadastro_p(cadastro dados[TAM]) {
	printf("Digite a quantidade de funcionarios que deseja cadastrar: \n");
	scanf("%d",&quant);
	for (i=0; i<quant; i++) {
		printf("Digite o nome: \n");

		printf("Digite o cargo: \n");
		scanf("%s",dados[i].cargo);

		printf("Digite o CPF:\n");
		scanf("%d",&dados[i].cpf);
		printf("***Cadastro realizado com sucesso***\n ");
	}
};
void exibe(cadastro dados[TAM]) {
	for(i=0; i<=quant-1; i++) {
		printf("_-_-_-_-_-_-_-_-_-_-_-_-_-_-_-_-_\n");
		printf("Funcionario %d:\n",i);
		printf("Nome: %s\n",dados[i].nome);
		printf("Cargo: %s \n",dados[i].cargo);
		printf("Salario: %s\n",dados[i].salario);
		printf("Idade: %d\n",dados[i].idade);
		printf("CPF: %d\n",dados[i].cpf);
		printf("_-_-_-_-_-_-_-_-_-_-_-_-_-_-_-_-_");
	}
};

void Quick(cadastro dados[TAM], int inicio, int fim ) {
	int pivo, i, j, meio;
	i=inicio;
	j=fim;

	meio=(i+j)/2;
	pivo=dados[meio].cpf;


	do {
		while(dados[i].cpf<pivo) {
			i++;
		}
		while(dados[j].cpf>pivo) {
			j--;
		}
		if(i<=j) {
			aux=dados[i];
			dados[i]=dados[j];
			dados[j]=aux;
			i++;
			j--;
		}
	} while(j>i);

	if(inicio<j) {
		Quick(dados,inicio,j);
	}
	if(i<fim) {
		Quick(dados,i,fim);
	}

};
